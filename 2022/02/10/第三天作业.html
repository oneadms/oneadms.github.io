 public class Work4 {
     public static void main(String[] args) {
         for (int i = 1000; i <= 9999; i++) {
             int sum = 0;
             int j=i;
             while (j != 0) {
                 int g = j% 10;
                 double gws = Math.pow(g, 4);

6.求出1-1000水仙花数的个数

水仙花数是指一个 3 位数，它的每个位上的数字的 3次幂之和等于它本身

public class Work5 {
    public static void main(String[] args) {
        for (int i = 100; i <= 1000; i++) {
            int temp = i;
            int sum = 0;
            while (temp != 0) {
                sum += Math.pow(temp % 10, 3);
                temp /= 10;
            }
            if (sum == i) {
                System.out.println(sum);
            }


        }
    }
}
